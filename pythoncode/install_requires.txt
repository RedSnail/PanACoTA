# install barrnap
wget https://github.com/tseemann/barrnap/archive/0.8.tar.gz
tar -xf 0.8.tar.gz
rm 0.8.tar.gz
cd barrnap-0.8
ln -s absolute_path/barrnap-0.8/bin/barrnap /usr/local/bin
